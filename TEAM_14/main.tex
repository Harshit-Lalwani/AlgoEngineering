\section{Problem Statement}
% ---

This project studies cache-aware and cache-oblivious algorithms with the goal of
designing and implementing efficient solutions for \textbf{String Sorting}.  We evaluate
these algorithms based on their ability to minimise I/O operations
$Q(N;\,M;\,B)$ between a fast internal memory of size $M$ and a larger memory,
where data is transferred in blocks of size $B$.  The primary metric of success
is how closely an algorithm approaches the scanning bound
$\text{Scan}(N) = O(N/B)$.

Table~\ref{tab:complexity} summarises the theoretical I/O and time complexity of
every algorithm studied.

  $N$: number of strings; $D = \sum_{i=1}^{N} d_i$: total distinguishing prefix
  length; $d_i$: distinguishing prefix length of string~$i$; $L$: burst
  threshold; $B$: block size; $M$: cache size; $\bar\ell$: mean string length.}
\label{tab:complexity}
\begin{tabular}{@{}llll@{}}
\toprule
Algorithm           & Time complexity        & I/O $Q(N;\,M;\,B)$                        & Cache model \\ \midrule

\section{The Architecture Problem with Baseline Algorithms}
% ---

Standard algorithms often ignore the memory hierarchy, leading to significant
performance degradation on large datasets.

\subsection{Standard Introsort (\texttt{std::sort})}

Standard implementations like \texttt{std::sort} treat strings as opaque objects.
When sorting $N$ strings, the algorithm performs $O(N \log N)$ comparisons.
In a string context, each comparison can trigger multiple cache misses as the CPU
fetches character data from arbitrary heap locations, leading to poor spatial
locality and high $Q(N)$.

\subsection{Multi-key Quicksort (MKQS)}

\textbf{Idea:} MKQS partitions strings into three sets (less than, equal to, and
greater than) based on the character at the current depth $d$.

\begin{algorithm}
\caption{Multi-key Quicksort ($A,\,lo,\,hi,\,d$)}
\begin{algorithmic}[1]
\If{$hi \leq lo$} \Return \EndIf
\State $pivot \gets \text{char\_at}(A[lo],\,d)$
\State Partition $A[lo \ldots hi]$ into $\{< pivot,\;= pivot,\;> pivot\}$
\State \textbf{MKQS}($A,\,lo,\,lt-1,\,d$)
\State \textbf{MKQS}($A,\,lt,\,gt,\,d+1$) \Comment{Advance depth for equal partition}
\State \textbf{MKQS}($A,\,gt+1,\,hi,\,d$)
\end{algorithmic}
\end{algorithm}

\textbf{Problem:} While MKQS avoids re-comparing characters, it suffers from poor
cache utilisation during partitioning.  If the dataset size $N > M$, the
random-access nature of the swaps causes frequent cache-line evictions.

\section{Cache-Aware Algorithms}
% ---

\subsection{Burstsort}

\textbf{Idea:} Burstsort uses a trie where leaves are small containers (buckets).
Each bucket is sized to fit inside the cache, so that the in-bucket sort always
operates within the fast memory.

\begin{algorithm}
\caption{Burstsort ($S,\,L$)}
\begin{algorithmic}[1]
\State Initialise trie root with an empty bucket
\For{each string $s \in S$}
    \State Traverse trie to the correct leaf bucket
    \State Insert $s$ into bucket
    \If{$\text{size(bucket)} > L$} \textbf{Burst}(bucket) \EndIf
\EndFor
\State Sort each bucket using a cache-efficient base-case sort (MKQS)
\end{algorithmic}
\end{algorithm}

\textbf{Complexity:}
\begin{itemize}
  \item \textbf{Time:} $O(D + N \log L)$, where $D = \sum_{i=1}^{N} d_i$ is the
        total distinguishing prefix length across all $N$ strings.
  \item \textbf{I/O:} $Q(N) \approx O\!\left(\tfrac{N}{B} \cdot \tfrac{D}{N \cdot L}\right)$.
        By choosing the burst threshold $L < M$, each bucket sort occurs entirely
        within the cache.  The dominant cost is then the initial trie distribution
        (a near-sequential scan) and the final readout, both $O(N/B)$.
\end{itemize}

\subsection{C-Burstsort}

C-Burstsort extends Burstsort with an arena-based memory allocator that replaces
per-string heap allocations with a single large contiguous arena.  This improves
spatial locality in two ways: (i)~the bucket pointer arrays are co-located in
memory, and (ii)~prefix characters are copied into contiguous key buffers, so the
in-bucket sort accesses a dense array rather than scattered heap pointers.  The
asymptotic I/O complexity matches Burstsort but the constant factor is
substantially lower in practice.

\subsection{MSD Radix Sort}

\textbf{Idea:} MSD Radix Sort uses a distribution-based approach, counting
frequencies of characters at depth $d$ and shuffling strings into $|\Sigma|$
buckets.

\begin{algorithm}
\caption{MSD Radix Sort ($A,\,lo,\,hi,\,d,\,aux$)}
\begin{algorithmic}[1]
\State \textbf{Count:} frequencies of $\text{char\_at}(s,d)$ for $s \in A[lo \ldots hi]$
\State \textbf{Indices:} transform counts into absolute start positions in $aux$
  \Comment{Invariant: $aux$ indexed from $lo$; write indices in $[lo, hi]$}
\State \textbf{Distribute:} move strings from $A$ to $aux[lo + \text{count}[c]]$ for each bucket $c$
\State \textbf{Copy back:} $A[lo \ldots hi] \gets aux[lo \ldots hi]$
\State \textbf{Recurse:} for each $r \in \Sigma$, sort sub-partition at depth $d+1$
\end{algorithmic}
\end{algorithm}

\textbf{Complexity:}
\begin{itemize}
  \item \textbf{I/O:} $Q(N) = O(D/B)$ when $|\Sigma| \le M/B$, where
        $D = \sum d_i$ is the total distinguishing prefix length.  Each pass is a
        $\text{Scan}(N)$ of cost $O(N/B)$; there are $\bar{d} = D/N$ passes on
        average.  If the alphabet size is large relative to $M/B$, the write
        pointers for distinct buckets reside in different cache lines and every
        write may cause a miss (\emph{bucket thrashing}).
  \item \textbf{Depth sensitivity:} Because the algorithm recurses once per
        distinguishing character, datasets with high mean LCP depth (e.g.\
        \texttt{wiki\_heavy}) incur proportionally more recursion levels and
        correspondingly higher total I/O.  This is directly measured in
        \texttt{radix\_depth.csv} and shown in Figure~\ref{fig:depth}.
\end{itemize}

\section{Cache-Oblivious Algorithm: Funnelsort (Lazy)}
% ---

\textbf{Idea:} Funnelsort uses a recursive divide-and-conquer approach that is
I/O-optimal without needing to know $M$ or $B$ at compile time.  The
implementation is a \emph{comparison-based} $K$-way merge sort applied to
string pointers (not a string-specific algorithm); $K = \lceil N^{1/3} \rceil$.

\begin{algorithm}
\caption{Lazy Funnelsort ($S$)}
\begin{algorithmic}[1]
\If{$|S| \le \text{THRESHOLD}$}
    \State Sort $S$ with insertion sort \Comment{THRESHOLD$\,{=}\,8192$; $8192 \times 14\,\text{B} \approx 114\,\text{KB} \le$ L2 per core}
\Else
    \State Divide $S$ into $K = \lceil|S|^{1/3}\rceil$ equal streams
    \State \textbf{Funnelsort}(each stream)
    \State Merge streams using a $K$-way \texttt{std::priority\_queue} merger
\EndIf
\end{algorithmic}
\end{algorithm}

\textbf{Complexity:}
\begin{itemize}
  \item \textbf{I/O:} $Q(N) = O\!\left(\tfrac{N}{B}\log_{M/B}\tfrac{N}{B}\right)$.
  \item \textbf{Proof sketch:} The recursion bottoms out when the subproblem
        satisfies $n \cdot \bar\ell \le M$, i.e.\ the string data and merge
        buffers fit entirely in cache.  All subsequent merges incur zero
        additional I/O.  This two-level locality argument holds for \emph{any}
        $M$ and $B$, giving the cache-oblivious guarantee.
  \item \textbf{Implementation note:} The present implementation is
        comparison-based on full strings rather than a string-specific
        cache-oblivious algorithm (e.g.\ signature sorting).  As a result,
        wall-clock time closely tracks \texttt{std::sort}; the advantage is the
        provably optimal I/O bound without knowing $M$ or $B$.
\end{itemize}

\section{Datasets}
% ---

Four datasets were used to stress distinct performance dimensions.

\subsection{Synthetic Datasets}
Generated with C++ pseudo-random generators for controlled variation.
\begin{itemize}
  \item \textbf{Random:} Strings of length 4--24, uniform character distribution.
        Tests base-case $N \log N$ behaviour with minimal LCP depth.
  \item \textbf{Prefix-Heavy:} Strings sharing a long common prefix
        (e.g.\ \texttt{cache\_aware\_1}, \texttt{cache\_aware\_2}).  Forces
        algorithms to process deep distinguishing prefixes, stressing $D$ in the
        complexity bounds.
\end{itemize}

\subsection{Real-World Dataset: Wikipedia Titles}
A collection of up to $5 \times 10^6$ titles from English Wikipedia.
Two variants are used:
\begin{itemize}
  \item \textbf{wiki\_random:} Titles drawn uniformly at random, representing a
        diverse vocabulary with moderate LCP depth.
  \item \textbf{wiki\_heavy:} Titles filtered to emphasise common prefixes
        (e.g.\ \texttt{List of\ldots}), approximating the long-tail prefix
        distribution of real encyclopaedic data and maximising $D$.
\end{itemize}

\subsection{Measurement Methodology}

For each combination of (algorithm, dataset, $N$), we execute \textbf{10
  independent trials} with distinct random seeds.  The first \textbf{2 runs are
  discarded} as warm-up to avoid cold-cache bias.  Wall-clock time is measured
with \texttt{std::chrono::steady\_clock} around the sort call only (dataset
generation is excluded).  Correctness is verified after every run via
\texttt{std::is\_sorted} with a lexicographic comparator.

Aggregate statistics reported are mean $\pm$ standard deviation across the 8
timed trials.  Cache-miss counts (L1 data misses, LLC data misses) are collected
at $N \in \{10^4,\,10^5,\,10^6\}$ using \texttt{perf stat} (Linux PMU) or
Valgrind/Cachegrind as a fallback, and written to \texttt{Analysis/cache\_results.csv}.

Problem sizes: $N \in \{10^5,\,10^6,\,5 \times 10^6\}$.

\subsection{Summary of Key Observations}

\begin{enumerate}
  \item \textbf{Cache-aware algorithms scale better.}
        Burstsort and C-Burstsort exhibit near-linear scaling in $N$ on all
        four datasets.  \texttt{std::sort} and MSD Radix Sort show super-linear
        growth as $N$ exceeds the LLC (12\,MiB on this machine).
  \item \textbf{Arena allocation matters.}
        C-Burstsort consistently outperforms the original Burstsort by a
        moderate constant factor, attributable to better spatial locality in
        arena-allocated key buffers.
  \item \textbf{MSD Radix Sort is depth-limited.}
        On \texttt{wiki\_heavy} and \texttt{prefix\_heavy}, mean recursion depth
        grows with $D/N$; this is directly correlated with time per character
        (Figure~\ref{fig:depth}) and explains MSD Radix Sort's outlier
        performance on these inputs.
  \item \textbf{Cache-oblivious Funnelsort tracks \texttt{std::sort}.}
        Because the implementation is comparison-based, its wall-clock time
        closely follows \texttt{std::sort}.  The key advantage is the provably
        optimal $O\!\left(\tfrac{N}{B}\log_{M/B}\tfrac{N}{B}\right)$ I/O
        bound without needing to know $M$ or $B$.
  \item \textbf{Architectural lesson.}
        For in-memory string sorting, engineering cache locality through
        bucket-size control ($L < M$) and contiguous memory layout (arena)
        delivers more practical gain than either a theoretically optimal
        cache-oblivious algorithm or a standard comparison/radix sort.
\end{enumerate}

\section{Codebase Architecture}
% ---

The project is implemented in \textbf{C++20} for high-performance memory
management and deterministic cache control.

\subsection{File Structure}
\begin{itemize}
  \item \texttt{main.cpp}: Entry point; manages dataset generation, loading, and
        the benchmarking loop (10 seeds, 2 warm-up runs, writes
        \texttt{results\_aggregated.csv} and \texttt{radix\_depth.csv}).
  \item \texttt{algorithms/}
    \begin{itemize}
      \item \texttt{radix.cpp}: MSD Radix Sort with correct absolute
            \texttt{aux} indexing.
      \item \texttt{burstsort.cpp}: Original Burstsort and arena-based
            C-Burstsort.
      \item \texttt{cache\_oblivious.cpp}: $K$-way Funnelsort
            ($K = \lceil N^{1/3} \rceil$, base-case threshold 8\,192 strings
            $\approx 114$\,KB $\le$ L2 per core on the test machine).
      \item \texttt{baselines.cpp}: MKQS and \texttt{std::sort} wrappers.
    \end{itemize}
  \item \texttt{Analysis/}
    \begin{itemize}
      \item \texttt{cache\_aware\_analysis.py}: Reads
            \texttt{results\_aggregated.csv}, \texttt{radix\_depth.csv}, and
            \texttt{cache\_results.csv}; writes summary CSVs and all figures.
      \item \texttt{run\_perf.sh}: Collects hardware cache counters via
            \texttt{perf stat} or Valgrind at
            $N \in \{10^4,\,10^5,\,10^6\}$.
    \end{itemize}
\end{itemize}
