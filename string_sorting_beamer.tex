\section{Problem \& Motivation}

%% ─── SLIDE 3: Problem Statement ────────────────────────────────────────────
\begin{frame}{Problem Statement}
  \begin{block}{Goal}
    Sort $N$ strings while \textbf{minimising I/O operations}
    $Q(N;\,M;\,B)$ between fast cache ($M$) and slower memory
    (block size $B$).
  \end{block}
  \vspace{0.5em}
  \begin{itemize}
    \item \textbf{Success metric:} approach the scan bound
          $\mathrm{Scan}(N) = O(N/B)$
    \item Modern CPUs: L1 48\,KiB, L2 1.25\,MiB, L3 12\,MiB
          $\Rightarrow$ memory hierarchy deeply affects runtime
    \item Standard sorts ignore cache $\Rightarrow$ $O(N \log N)$
          cache misses on large $N$
  \end{itemize}
  \vspace{0.5em}
  \begin{alertblock}{Key insight}
    Engineering cache locality delivers more practical speedup than
    asymptotic optimality alone.
  \end{alertblock}
\end{frame}

%% ─── SLIDE 4: I/O Model ────────────────────────────────────────────────────
\begin{frame}{Memory Hierarchy \& I/O Model}
  \begin{columns}
    \column{0.52\textwidth}
      \textbf{External-memory (EM) model}
      \begin{itemize}
        \item Two levels: fast cache of size $M$, slow disk
        \item Data transferred in blocks of size $B$
        \item Cost $=$ number of block transfers $Q(N;M;B)$
      \end{itemize}
      \vspace{0.5em}
      \textbf{Cache-oblivious model}
      \begin{itemize}
        \item Same two-level structure but algorithm \emph{does not}
              know $M$ or $B$
        \item Optimal for all levels simultaneously
      \end{itemize}
    \column{0.44\textwidth}
      \begin{block}{Test machine (WSL2)}
        \begin{tabular}{ll}
          CPU & i5-12450H \\
          L1d & 48\,KiB/core \\
          L2  & 1.25\,MiB/core \\
          LLC & 12\,MiB \\
          RAM & 7.6\,GiB \\
          OS  & kernel 6.6.87 \\
        \end{tabular}
      \end{block}
  \end{columns}
\end{frame}

\section{Algorithms}

%% ─── SLIDE 5: Complexity Summary ───────────────────────────────────────────
\begin{frame}{Theoretical Complexity Summary}
  \footnotesize
  \begin{tabular}{@{}lllc@{}}
    \toprule
    Algorithm & Time & I/O $Q(N;M;B)$ & Model \\
    \midrule
    \texttt{std::sort} & $O(N\log N\cdot\bar\ell)$
      & $O\!\left(\tfrac{N}{B}\log N\right)$ & agnostic \\
    Multikey QS & $O(D+N\log N)$
      & $O\!\left(\tfrac{N}{B}\log N\right)$ & agnostic \\
    MSD Radix & $O(D)$
      & $O(D/B)$ if $|\Sigma|\le M/B$ & aware \\
    Burstsort & $O(D+N\log L)$
      & $O\!\left(\tfrac{N}{B}\cdot\tfrac{D}{NL}\right)$ & aware \\
    C-Burstsort & $O(D+N\log L)$
      & $O\!\left(\tfrac{N}{B}\cdot\tfrac{D}{NL}\right)$ & aware \\
    Funnelsort & $O(N\log N)$
      & $O\!\left(\tfrac{N}{B}\log_{M/B}\tfrac{N}{B}\right)$ & oblivious \\
    \bottomrule
  \end{tabular}
  \vspace{0.4em}
  \scriptsize
  $D=\sum d_i$: total distinguishing prefix length;
  $L$: burst threshold;
  $\bar\ell$: mean string length.
\end{frame}

%% ─── SLIDE 6: Burstsort ────────────────────────────────────────────────────
\begin{frame}{Cache-Aware: Burstsort \& C-Burstsort}
  \begin{columns}
    \column{0.55\textwidth}
      \textbf{Burstsort idea:}
      \begin{itemize}
        \item Trie with leaf \emph{buckets} sized $L < M$
        \item In-bucket sort stays entirely in cache
        \item I/O dominated by trie traversal: near $O(N/B)$
      \end{itemize}
      \vspace{0.5em}
      \textbf{C-Burstsort (arena variant):}
      \begin{itemize}
        \item Single contiguous arena replaces heap allocation
        \item Prefix chars copied to dense key buffers
        \item Same $O$ complexity, substantially lower constant
      \end{itemize}
    \column{0.42\textwidth}
      \begin{block}{I/O bound}
        \[ Q \approx O\!\left(\frac{N}{B}\cdot\frac{D}{N\cdot L}\right) \]
        Choose $L < M$: bucket sorts free; trie scan $= O(N/B)$
      \end{block}
  \end{columns}
\end{frame}

%% ─── SLIDE 7: MSD Radix ────────────────────────────────────────────────────
\begin{frame}{Cache-Aware: MSD Radix Sort}
  \begin{itemize}
    \item Count frequencies at depth $d$, distribute into $|\Sigma|$ buckets,
          recurse per bucket
    \item I/O: $O(D/B)$ when $|\Sigma|\le M/B$ — each pass is a scan
    \item \textbf{Depth sensitivity:} mean recursion depth $= D/N$;
          high-LCP datasets (\texttt{wiki\_heavy}) cause many passes
    \item \alert{Bucket thrashing} when $|\Sigma| > M/B$: every write hits a
          different cache line
  \end{itemize}
  \vspace{0.3em}
  \begin{alertblock}{Fixed in implementation}
    Distribute writes at \texttt{lo + count[c]} (not absolute 0);
    copy-back uses the same $[lo, hi]$ window.
  \end{alertblock}
\end{frame}

%% ─── SLIDE 8: Funnelsort ───────────────────────────────────────────────────
\begin{frame}{Cache-Oblivious: Lazy Funnelsort}
  \begin{columns}
    \column{0.56\textwidth}
      \begin{itemize}
        \item $K$-way merge sort, $K = \lceil N^{1/3}\rceil$
        \item Base case: 8\,192 strings
              ($\approx 114$\,KB $\le$ L2 per core)
        \item Proof sketch: recursion bottoms out when
              $n\cdot\bar\ell \le M$;
              subsequent merges incur \emph{zero} extra I/O
        \item Holds for \emph{any} $M$ and $B$
      \end{itemize}
    \column{0.41\textwidth}
      \begin{block}{Optimal I/O}
        \[ Q = O\!\left(\frac{N}{B}\log_{M/B}\frac{N}{B}\right) \]
      \end{block}
      \begin{block}{Limitation}
        Comparison-based $\Rightarrow$
        wall-clock $\approx$ \texttt{std::sort};
        advantage is provable I/O bound
        without knowing $M$, $B$.
      \end{block}
  \end{columns}
\end{frame}

\section{Datasets}

%% ─── SLIDE 9: Datasets ─────────────────────────────────────────────────────
\begin{frame}{Four Datasets}
  \begin{columns}
    \column{0.48\textwidth}
      \textbf{Synthetic}
      \begin{itemize}
        \item \textbf{random} — length 4--24, uniform chars;
              minimal LCP depth; tests $N\log N$ base case
        \item \textbf{prefix\_heavy} — shared long common prefix;
              stresses $D$; exposes MSD Radix thrashing
      \end{itemize}
    \column{0.48\textwidth}
      \textbf{Wikipedia titles (up to $5\times10^6$)}
      \begin{itemize}
        \item \textbf{wiki\_random} — uniform sample;
              moderate LCP; representative real-world vocabulary
        \item \textbf{wiki\_heavy} — ``List of\ldots'' filtered;
              maximises $D$; worst case for depth-sensitive
              algorithms
      \end{itemize}
  \end{columns}
  \vspace{0.4em}
  \begin{alertblock}{Why four datasets?}
    Each dimension isolates a different complexity parameter:
    $N$, $D/N$, $|\Sigma|$, and real-world prefix distribution.
  \end{alertblock}
\end{frame}

\begin{frame}{Experimental Setup}
  \begin{columns}
    \column{0.52\textwidth}
      \textbf{Methodology}
      \begin{itemize}
        \item 10 seeds per (algorithm, dataset, $N$)
        \item 2 warm-up runs discarded (cold-cache bias)
        \item Timer: \texttt{std::chrono::steady\_clock}
              around sort only
        \item Correctness: \texttt{std::is\_sorted} after every run
        \item Cache counters: \texttt{perf stat} at
              $N\in\{10^4,10^5,10^6\}$
        \item CPU pinned: \texttt{taskset -c 0},
              frequency scaling off

\section{Key Takeaways}

%% ─── SLIDE 17: Key Observations ────────────────────────────────────────────
\begin{frame}{Key Observations}
  \begin{enumerate}
    \item \textbf{Cache-aware $\succ$ cache-agnostic at large $N$.}
          Burstsort/C-Burstsort near-linear; \texttt{std::sort} super-linear
          beyond LLC (12\,MiB here).
    \item \textbf{Arena allocation matters.}
          C-Burstsort beats Burstsort by a consistent constant factor
          (dense key buffers = better spatial locality).
    \item \textbf{MSD Radix is depth-limited.}
          Recursion depth $\propto D/N$; directly causes outlier cost
          on \texttt{wiki\_heavy} and \texttt{prefix\_heavy}.
    \item \textbf{Funnelsort $\approx$ \texttt{std::sort} in wall-clock.}
          Comparison-based design; value is provably optimal I/O
          without knowing $M$ or $B$.
    \item \textbf{Architectural lesson.}
          Bucket-size control ($L < M$) $+$ contiguous layout (arena)
          $>$ theoretical optimality alone.
  \end{enumerate}
\end{frame}
